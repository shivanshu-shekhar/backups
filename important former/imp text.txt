import React, { Component } from 'react';
import axios from 'axios';
import { Link } from "react-router-dom";

class MensClothing extends Component {
  state = {
    isLoading: true,
    isError: false,
    detail: []
  };
  componentDidMount() {
    axios
      .get('http://localhost:5000/data?type=men')
      .then((res) => {
        console.log(res.data);
        //TODO: make use of above data and display the ui
        this.setState({
          isLoading: false,
          isError: false,
          detail: res.data,
        });
      })
      .catch((err) => {
        console.log(err);
        this.setState({
          isLoading: false,
          isError: true
        });
      })
      .finally(() => {
        console.log('It is over!');
      });
  }
  render() {
    if (this.state.isLoading) {
      return (
        <div className="text-center mt-5">
          <div className="spinner-border text-danger" role="status"></div>
        </div>
      );
    }
    //search for error
    if (this.state.isError) {
      return (
        <div className="alert alert-danger" role="alert">
          Sorry unable to fetch photos. Try after sometime.
        </div>
      );
    }
    return (
      <div>
        <div className="row pt-3">
          {this.state.detail.map(({ id, url, title, alt}) => (
            <div className="col-md-3" key={id}>
              <div className="card shadow-sm">
                <img src={url} alt={alt} />
                <div className="card-body">
                  <p className="card-text"><Link to="products-details">{title}</Link></p>
                </div>
              </div>
            </div>
          ))}
        </div>
      </div>
    )
  }
}
export default MensClothing;
